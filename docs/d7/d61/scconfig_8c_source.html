58u32 FindVirtionetDev() {
59 for (u8 bus = 0; bus < 255; bus++) {
60 for (u8 slot = 0; slot < 32; slot++) {
61 for (u8 func = 0; func < 8; func++) {
62 u16 vendor = PciConfigRead(bus, slot, func, 0x00) & 0xFFFF;
63 if (vendor == 0xFFFF) {
64 continue;
65 }
66 u16 dev = (PciConfigRead(bus, slot, func, 0x00) >> 16) & 0xFFFF;
67 bar0 = PciConfigRead(bus, slot, func, 0x10);
68 u32 iob = bar0 & ~0xF;
69 u32 revid = (PciConfigRead(bus, slot, func, 0x08) >> 0) & 0xFF;
70
71 if (vendor == 0x1AF4 && dev == 0x1000 && revid == 0x00) {
72 virtio_bus = bus;
73 virtio_slot = slot;
74 virtio_func = func;
75 Kprintf("VirtIO at %x:%x.%x BAR0=%x IO=%x Rev=%x\n", bus,
76 slot, func, bar0, iob, revid);
77
78 // 1. Reset device
79 Outb(VIRTIO_REG(0x12), 0x00);
80
81 // 2. Set ACKNOWLEDGE and DRIVER status bits
82 Outb(VIRTIO_REG(0x12), 0x01); // AcKnowledge
83 Outb(VIRTIO_REG(0x12), 0x03); // AcKnowledge | Driver
84
85 // 3. Read device features
86 u32 features = Inl(VIRTIO_REG(0x00));
87 // Optionally masK features here
88
89 // 4. Write accepted features
90 Outl(VIRTIO_REG(0x04), features); // For now, accept all
91
92 // 5. Set DRIVER_OK bit
93 Outb(VIRTIO_REG(0x12),
94 0x07); // AcKnowledge | Driver | Features OK
95 return iob;
96 }
97 }
98 }
99 }
100 return (u32)(uptr)NULL;
