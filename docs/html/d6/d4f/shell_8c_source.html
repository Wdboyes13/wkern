151void Sh() {
152 while (1) {
153 char *cmd = (char *)Kmalloc(128, 8);
154 if (!cmd) {
155 Kprintf("Malloc Failed");
156 Kfree(cmd);
157 continue;
158 }
159
160 Kprintf("\nSH> ");
161 Kgetstr(cmd, 127);
162 Kflush();
163
164 char *argv[8];
165 int argc = Split(cmd, argv, 8);
166
167 if (argc == 0) {
168 Kfree(cmd);
169 continue;
170 }
171
172 int found = 0;
173 for (int i = 0; i < NUM_COMMANDS; i++) {
174 if (Kstrcmp(argv[0], commands[i].name) == 0) {
175 commands[i].func((const char **)argv, argc);
176 found = 1;
177 break;
178 }
179 }
180
181 if (!found) {
182 Kprintf("\nUnKnown Command - Try `help`\n");
183 }
184
185 Kfree(cmd);
186 }
187}
