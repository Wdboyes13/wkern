40void Mkfile(const char *filename, const char *ext) {
41 int free = -1;
42 int sector_to_write = -1;
43 u32 entries_per_sector = fat16.bytes_per_sector / 32;
44 u32 root_dir_sectors =
45 ((fat16.root_entry_count * 32) + (fat16.bytes_per_sector - 1)) /
46 fat16.bytes_per_sector;
47
48 u8 sector[512];
49 u16 clust = 0;
50
51 // Find free root directory entry
52 for (u32 i = 0; i < root_dir_sectors; i++) {
53 AtaReadSector(fat16.root_dir_start_lba + i, sector);
54 for (u32 j = 0; j < entries_per_sector; j++) {
55 u8 *entry = &sector[j * 32];
56 if (entry[0] == 0x00 || entry[0] == 0xE5) {
57 free = j;
58 sector_to_write = i;
59 goto found_free_entry; // breaK out of both loops ASAP
60 }
61 }
62 }
63
64found_free_entry:
65 if (free == -1 || sector_to_write == -1) {
66 Kprintf("No free root directory entry found!\n");
67 return;
68 }
69
70 // Find free cluster
71 for (u16 c = 2; c < fat16.total_cluster; c++) {
72 u32 fat_offset = c * 2;
73 u32 fat_sector = fat16.fat_start_lba + (fat_offset / 512);
74 u32 fat_index = fat_offset % 512;
75
76 u8 fatbuf[512];
77 AtaReadSector(fat_sector, fatbuf);
78
79 if (fatbuf[fat_index] == 0x00 && fatbuf[fat_index + 1] == 0x00) {
80 clust = c;
81 fatbuf[fat_index] = 0xFF;
82 fatbuf[fat_index + 1] = 0xFF;
83 AtaWriteSector(fat_sector, fatbuf);
84 break;
85 }
86 }
87
88 if (clust == 0) {
89 Kprintf("No free cluster found!\n");
90 return;
91 }
92
93 // Now write the root directory entry
94 AtaReadSector(fat16.root_dir_start_lba + sector_to_write,
95 sector); // Re-read sector for safety
96 u8 *entry = &sector[free * 32];
97
98 // Clear entry bytes (optional but good practice)
99 for (int i = 0; i < 32; i++) {
100 entry[i] = 0x00;
101 }
102
103 char name_pad[8];
104 char ext_pad[3];
105 Padname(filename, name_pad, 8);
106 Padname(ext, ext_pad, 3);
107 Kmemcpy(entry + 0x00, name_pad, 8);
108 Kmemcpy(entry + 0x08, ext_pad, 3);
109 entry[0x0B] = 0x20; // Attribute: Archive
110
111 // Time and Date (optional, set 0 here)
112 entry[0x0E] = 0;
113 entry[0x0F] = 0;
114 entry[0x10] = 0;
115 entry[0x11] = 0;
116
117 // First cluster (little endian)
118 entry[0x1A] = clust & 0xFF;
119 entry[0x1B] = (clust >> 8) & 0xFF;
120
121 // File size = 0 for new file
122 entry[0x1C] = 0;
123 entry[0x1D] = 0;
124 entry[0x1E] = 0;
125 entry[0x1F] = 0;
126
127 AtaWriteSector(fat16.root_dir_start_lba + sector_to_write, sector);
128 Kprintf("File entry written successfully.\n");
129}
