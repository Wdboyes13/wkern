41void Fileconts(const char *filename, const char *ext) {
42 u32 entries_per_sector = fat16.bytes_per_sector / 32;
43 u32 root_dir_sectors =
44 ((fat16.root_entry_count * 32) + (fat16.bytes_per_sector - 1)) /
45 fat16.bytes_per_sector;
46
47 u8 sector[512];
48
49 for (u32 i = 0; i < root_dir_sectors; i++) {
50 AtaReadSector(fat16.root_dir_start_lba + i, sector);
51 for (u32 j = 0; j < entries_per_sector; j++) {
52 // Padding entries to be exactly 8 and 3, and upercased (blame msft
53 // for FAT16)
54 u8 *entry = &sector[j * 32];
55 char name_pad[8];
56 char ext_pad[3];
57 Padname(filename, name_pad, 8);
58 Padname(ext, ext_pad, 3);
59
60 if (entry[0] != 0x00 &&
61 entry[0] != 0xE5 && // ChecK if entry is not deleted & exists
62 Kmemcmp(entry + 0x00, name_pad, 8) == 0 && // Compare name
63 Kmemcmp(entry + 0x08, ext_pad, 3) == 0) { // Compare ext
64 u16 clust = entry[0x1A] | (entry[0x1B] << 8);
65 if (clust < 2) {
66 Kprintf("Invalid Cluster\n");
67 return;
68 }
69 u32 lba = fat16.data_start_lba +
70 ((clust - 2) * fat16.sectors_per_cluster);
71
72 u8 data[512] = {0};
73 AtaReadSector(lba, data); // Read data in the FAT16 Sector
74 for (int i = 0; i < 512; i++) {
75 if (data[i] == 0x1A) {
76 break; // EOF
77 }
78 if (data[i] >= 0x20 && data[i] <= 0x7E) {
79 Kputchar(data[i]); // printable ASCII
80 } else {
81 Kputchar('.'); // non-printable gets a dot
82 }
83 }
84 Kputchar('\n');
85 }
86 }
