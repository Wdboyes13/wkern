86static int set_len(const char *re, int re_len) {
87 int len = 0;
88
89 while (len < re_len && re[len] != ']') {
90 len += op_len(re + len);

110static int match_op(const unsigned char *re, const unsigned char *s,
111 struct regex_info *info) {
112 int result = 0;
113 switch (*re) {
114 case '\\':
115 /* Metacharacters */
116 switch (re[1]) {
117 case 'S':
118 FAIL_IF(Kisspace(*s), SLRE_NO_MATCH);
119 result++;
120 break;
121 case 's':
122 FAIL_IF(!Kisspace(*s), SLRE_NO_MATCH);
123 result++;
124 break;
125 case 'd':
126 FAIL_IF(!Kisdigit(*s), SLRE_NO_MATCH);
127 result++;
128 break;
129 case 'b':
130 FAIL_IF(*s != '\b', SLRE_NO_MATCH);
131 result++;
132 break;
133 case 'f':
134 FAIL_IF(*s != '\f', SLRE_NO_MATCH);
135 result++;
136 break;
137 case 'n':
138 FAIL_IF(*s != '\n', SLRE_NO_MATCH);
139 result++;
140 break;
141 case 'r':
142 FAIL_IF(*s != '\r', SLRE_NO_MATCH);
143 result++;
144 break;
145 case 't':
146 FAIL_IF(*s != '\t', SLRE_NO_MATCH);
147 result++;
148 break;
149 case 'v':
150 FAIL_IF(*s != '\v', SLRE_NO_MATCH);
151 result++;
152 break;
153
154 case 'x':
155 /* Match byte, \xHH where HH is hexadecimal byte representaion */
156 FAIL_IF(hextoi(re + 2) != *s, SLRE_NO_MATCH);
157 result++;
158 break;
159
160 default:
161 /* Valid metacharacter check is done in bar() */
162 FAIL_IF(re[1] != s[0], SLRE_NO_MATCH);
163 result++;
164 break;
165 }
166 break;
167
168 case '|':
169 FAIL_IF(1, SLRE_INTERNAL_ERROR);
170 break;
171 case '$':
172 FAIL_IF(1, SLRE_NO_MATCH);
173 break;
174 case '.':
175 result++;
176 break;
177
178 default:
179 if (info->flags & SLRE_IGNORE_CASE) {
180 FAIL_IF(Ktolower(*re) != Ktolower(*s), SLRE_NO_MATCH);
181 } else {
182 FAIL_IF(*re != *s, SLRE_NO_MATCH);
183 }
184 result++;
185 break;
186 }
187
188 return result;
189}

191static int match_set(const char *re, int re_len, const char *s,
192 struct regex_info *info) {
193 int len = 0, result = -1, invert = re[0] == '^';
194
195 if (invert)
196 re++, re_len--;
197
198 while (len <= re_len && re[len] != ']' && result <= 0) {
199 /* Support character range */
200 if (re[len] != '-' && re[len + 1] == '-' && re[len + 2] != ']' &&
201 re[len + 2] != '\0') {
202 result = info->flags & SLRE_IGNORE_CASE
203 ? Ktolower(*s) >= Ktolower(re[len]) &&
204 Ktolower(*s) <= Ktolower(re[len + 2])
205 : *s >= re[len] && *s <= re[len + 2];
206 len += 3;
207 } else {
208 result = match_op((const unsigned char *)re + len,
209 (const unsigned char *)s, info);
210 len += op_len(re + len);
211 }
212 }
213 return (!invert && result > 0) || (invert && result <= 0) ? 1 : -1;
214}

218static int bar(const char *re, int re_len, const char *s, int s_len,
219 struct regex_info *info, int bi) {
220 /* i is offset in re, j is offset in s, bi is brackets index */
221 int i, j, n, step;
222
223 for (i = j = 0; i < re_len && j <= s_len; i += step) {
224
225 /* Handle quantifiers. Get the length of the chunk. */
226 step = re[i] == '(' ? info->brackets[bi + 1].len + 2
227 : get_op_len(re + i, re_len - i);
228
229 DBG(("%s [%.*s] [%.*s] re_len=%d step=%d i=%d j=%d\n", __func__,
230 re_len - i, re + i, s_len - j, s + j, re_len, step, i, j));
231
232 FAIL_IF(is_quantifier(&re[i]), SLRE_UNEXPECTED_QUANTIFIER);
233 FAIL_IF(step <= 0, SLRE_INVALID_CHARACTER_SET);
234
235 if (i + step < re_len && is_quantifier(re + i + step)) {
236 DBG(("QUANTIFIER: [%.*s]%c [%.*s]\n", step, re + i, re[i + step],
237 s_len - j, s + j));
238 if (re[i + step] == '?') {
239 int result = bar(re + i, step, s + j, s_len - j, info, bi);
240 j += result > 0 ? result : 0;
241 i++;
242 } else if (re[i + step] == '+' || re[i + step] == '*') {
243 int j2 = j, nj = j, n1, n2 = -1, ni, non_greedy = 0;
244
245 /* Points to the regexp code after the quantifier */
246 ni = i + step + 1;
247 if (ni < re_len && re[ni] == '?') {
248 non_greedy = 1;
249 ni++;
250 }
251
252 do {
253 if ((n1 = bar(re + i, step, s + j2, s_len - j2, info, bi)) >
254 0) {
255 j2 += n1;
256 }
257 if (re[i + step] == '+' && n1 < 0)
258 break;
259
260 if (ni >= re_len) {
261 /* After quantifier, there is nothing */
262 nj = j2;
263 } else if ((n2 = bar(re + ni, re_len - ni, s + j2,
264 s_len - j2, info, bi)) >= 0) {
265 /* Regex after quantifier matched */
266 nj = j2 + n2;
267 }
268 if (nj > j && non_greedy)
269 break;
270 } while (n1 > 0);
271
272 /*
273 * Even if we found one or more pattern, this branch will be
274 * executed, changing the next captures.
275 */
276 if (n1 < 0 && n2 < 0 && re[i + step] == '*' &&
277 (n2 = bar(re + ni, re_len - ni, s + j, s_len - j, info,
278 bi)) > 0) {
279 nj = j + n2;
280 }
281
282 DBG(("STAR/PLUS END: %d %d %d %d %d\n", j, nj, re_len - ni, n1,
283 n2));
284 FAIL_IF(re[i + step] == '+' && nj == j, SLRE_NO_MATCH);
285
286 /* If while loop body above was not executed for the *
287 * quantifier, */
288 /* make sure the rest of the regex matches */
289 FAIL_IF(nj == j && ni < re_len && n2 < 0, SLRE_NO_MATCH);
290
291 /* Returning here cause we've matched the rest of RE already */
292 return nj;
293 }
294 continue;
295 }
296
297 if (re[i] == '[') {
298 n = match_set(re + i + 1, re_len - (i + 2), s + j, info);
299 DBG(("SET %.*s [%.*s] -> %d\n", step, re + i, s_len - j, s + j, n));
300 FAIL_IF(n <= 0, SLRE_NO_MATCH);
301 j += n;
302 } else if (re[i] == '(') {
303 n = SLRE_NO_MATCH;
304 bi++;
305 FAIL_IF(bi >= info->num_brackets, SLRE_INTERNAL_ERROR);
306 DBG(("CAPTURING [%.*s] [%.*s] [%s]\n", step, re + i, s_len - j,
307 s + j, re + i + step));
308
309 if (re_len - (i + step) <= 0) {
310 /* Nothing follows brackets */
311 n = doh(s + j, s_len - j, info, bi);
312 } else {
313 int j2;
314 for (j2 = 0; j2 <= s_len - j; j2++) {
315 if ((n = doh(s + j, s_len - (j + j2), info, bi)) >= 0 &&
316 bar(re + i + step, re_len - (i + step), s + j + n,
317 s_len - (j + n), info, bi) >= 0)
318 break;
319 }
320 }
321
322 DBG(("CAPTURED [%.*s] [%.*s]:%d\n", step, re + i, s_len - j, s + j,
323 n));
324 FAIL_IF(n < 0, n);
325 if (info->caps != NULL && n > 0) {
326 info->caps[bi - 1].ptr = s + j;
327 info->caps[bi - 1].len = n;
328 }
329 j += n;
330 } else if (re[i] == '^') {
331 FAIL_IF(j != 0, SLRE_NO_MATCH);
332 } else if (re[i] == '$') {
333 FAIL_IF(j != s_len, SLRE_NO_MATCH);
334 } else {
335 FAIL_IF(j >= s_len, SLRE_NO_MATCH);
336 n = match_op((const unsigned char *)(re + i),
337 (const unsigned char *)(s + j), info);
338 FAIL_IF(n <= 0, n);
339 j += n;
340 }
341 }
342
343 return j;
344}

346/* Process branch points */
347static int doh(const char *s, int s_len, struct regex_info *info, int bi) {
348 const struct bracket_pair *b = &info->brackets[bi];
349 int i = 0, len, result;
350 const char *p;
351
352 do {
353 p = i == 0 ? b->ptr : info->branches[b->branches + i - 1].schlong + 1;
354 len = b->num_branches == 0 ? b->len
355 : i == b->num_branches
356 ? (int)(b->ptr + b->len - p)
357 : (int)(info->branches[b->branches + i].schlong - p);
358 DBG(("%s %d %d [%.*s] [%.*s]\n", __func__, bi, i, len, p, s_len, s));
359 result = bar(p, len, s, s_len, info, bi);
360 DBG(("%s <- %d\n", __func__, result));
361 } while (result <= 0 && i++ < b->num_branches); /* At least 1 iteration */
362
363 return result;
364}

366static int baz(const char *s, int s_len, struct regex_info *info) {
367 int i, result = -1, is_anchored = info->brackets[0].ptr[0] == '^';
368
369 for (i = 0; i <= s_len; i++) {
370 result = doh(s + i, s_len - i, info, 0);
371 if (result >= 0) {
372 result += i;
373 break;
374 }
375 if (is_anchored)
376 break;
377 }
378
379 return result;
380}

382static void setup_branch_points(struct regex_info *info) {
383 int i, j;
384 struct branch tmp;
385
386 /* First, sort branches. Must be stable, no qsort. Use bubble algo. */
387 for (i = 0; i < info->num_branches; i++) {
388 for (j = i + 1; j < info->num_branches; j++) {
389 if (info->branches[i].bracket_index >
390 info->branches[j].bracket_index) {
391 tmp = info->branches[i];
392 info->branches[i] = info->branches[j];
393 info->branches[j] = tmp;
394 }
395 }
396 }
397
398 /*
399 * For each bracket, set their branch points. This way, for every bracket
400 * (i.e. every chunk of regex) we know all branch points before matching.
401 */
402 for (i = j = 0; i < info->num_brackets; i++) {
403 info->brackets[i].num_branches = 0;
404 info->brackets[i].branches = j;
405 while (j < info->num_branches && info->branches[j].bracket_index == i) {
406 info->brackets[i].num_branches++;
407 j++;
408 }
409 }
410}

412static int foo(const char *re, int re_len, const char *s, int s_len,
413 struct regex_info *info) {
414 int i, step, depth = 0;
415
416 /* First bracket captures everything */
417 info->brackets[0].ptr = re;
418 info->brackets[0].len = re_len;
419 info->num_brackets = 1;
420
421 /* Make a single pass over regex string, memorize brackets and branches */
422 for (i = 0; i < re_len; i += step) {
423 step = get_op_len(re + i, re_len - i);
424
425 if (re[i] == '|') {
426 FAIL_IF(info->num_branches >= (int)ARRAY_SIZE(info->branches),
427 SLRE_TOO_MANY_BRANCHES);
428 info->branches[info->num_branches].bracket_index =
429 info->brackets[info->num_brackets - 1].len == -1
430 ? info->num_brackets - 1
431 : depth;
432 info->branches[info->num_branches].schlong = &re[i];
433 info->num_branches++;
434 } else if (re[i] == '\\') {
435 FAIL_IF(i >= re_len - 1, SLRE_INVALID_METACHARACTER);
436 if (re[i + 1] == 'x') {
437 /* Hex digit specification must follow */
438 FAIL_IF(re[i + 1] == 'x' && i >= re_len - 3,
439 SLRE_INVALID_METACHARACTER);
440 FAIL_IF(re[i + 1] == 'x' &&
441 !(Kisxdigit(re[i + 2]) && Kisxdigit(re[i + 3])),
442 SLRE_INVALID_METACHARACTER);
443 } else {
444 FAIL_IF(!is_metacharacter((const unsigned char *)re + i + 1),
445 SLRE_INVALID_METACHARACTER);
446 }
447 } else if (re[i] == '(') {
448 FAIL_IF(info->num_brackets >= (int)ARRAY_SIZE(info->brackets),
449 SLRE_TOO_MANY_BRACKETS);
450 depth++; /* Order is important here. Depth increments first. */
451 info->brackets[info->num_brackets].ptr = re + i + 1;
452 info->brackets[info->num_brackets].len = -1;
453 info->num_brackets++;
454 FAIL_IF(info->num_caps > 0 &&
455 info->num_brackets - 1 > info->num_caps,
456 SLRE_CAPS_ARRAY_TOO_SMALL);
457 } else if (re[i] == ')') {
458 int ind = info->brackets[info->num_brackets - 1].len == -1
459 ? info->num_brackets - 1
460 : depth;
461 info->brackets[ind].len = (int)(&re[i] - info->brackets[ind].ptr);
462 DBG(("SETTING BRACKET %d [%.*s]\n", ind, info->brackets[ind].len,
463 info->brackets[ind].ptr));
464 depth--;
465 FAIL_IF(depth < 0, SLRE_UNBALANCED_BRACKETS);
466 FAIL_IF(i > 0 && re[i - 1] == '(', SLRE_NO_MATCH);
467 }
468 }
469
470 FAIL_IF(depth != 0, SLRE_UNBALANCED_BRACKETS);
471 setup_branch_points(info);
472
473 return baz(s, s_len, info);
474}

476int slre_match(const char *regexp, const char *s, int s_len,
477 struct slre_cap *caps, int num_caps, int flags) {
478 struct regex_info info;
479
480 /* Initialize info structure */

482 info.num_brackets = info.num_branches = 0;
483 info.num_caps = num_caps;
