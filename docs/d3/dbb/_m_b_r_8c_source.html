69u32 FindFat16Partition() {

71 Kprintf("Read ATA Sector\n");
72
73 struct MBR *mbr = (struct MBR *)sector;
74
75 if (mbr->signature != 0xAA55) {

78
79 Kprintf("Valid MBR Signature 0xAA55 Detected\n");
80
81 for (int i = 0; i < 4; i++) {
82 struct PartitionEntry *part = &mbr->partitions[i];
83 Kprintf("Partition type: 0x%x %x\n", i, part->type);
84 Kprintf(", LBA start: %x\n", part->lba_first_sector);
85
86 Kprintf("Partition dump: %x", i);

91 Kputchar('\n');
92
