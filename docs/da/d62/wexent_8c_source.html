50int chKwex(const unsigned char *smt, int smt_len, const unsigned char *st,
51 int st_len) {
52 for (int i = 0; i < smt_len; i++) {
53 int valid = 0;
54 for (int j = 0; j < st_len; j++) {
55 if (smt[i] == st[j]) {
56 valid = 1;
57 break;
58 }
59 }
60 if (!valid) {
61 return 1; // Invalid symbol found
62 }
63 }
64 return 0; // All symbols valid
65}
