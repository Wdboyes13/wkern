21static char *slre_replace(const char *regex, const char *buf,
22 const char *sub) {
23 char *s = NULL;
24 int n, n1, n2, n3, s_len, len = strlen(buf);
25 struct slre_cap cap = { NULL, 0 };
26
27 do {
28 s_len = s == NULL ? 0 : strlen(s);
29 if ((n = slre_match(regex, buf, len, &cap, 1, 0)) > 0) {
30 n1 = cap.ptr - buf, n2 = strlen(sub),
31 n3 = &buf[n] - &cap.ptr[cap.len];
32 } else {
33 n1 = len, n2 = 0, n3 = 0;
34 }
35 s = (char *) realloc(s, s_len + n1 + n2 + n3 + 1);
36 memcpy(s + s_len, buf, n1);
37 memcpy(s + s_len + n1, sub, n2);
38 memcpy(s + s_len + n1 + n2, cap.ptr + cap.len, n3);
39 s[s_len + n1 + n2 + n3] = '\0';
40
41 buf += n > 0 ? n : len;
42 len -= n > 0 ? n : len;
43 } while (len > 0);
44
45 return s;
46}
