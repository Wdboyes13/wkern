34void AtaPoll() {
35 for (int i = 0; i < 4; i++) {
36 Inb(ATA_CTRL); // 400ns delay
37 }
38
39 while (Inb(ATA_STATUS) & ATA_SR_BSY) {
40 ; // Wait for BSY to clear
41 }
42 u8 status = Inb(ATA_STATUS);
43 if (status & ATA_SR_ERR) {
44 Panic("ATA ERR in poll");
45 }
46 if (!(status & ATA_SR_DRQ)) {
47 Panic("ATA DRQ not set in poll");
48 }
49}

63void AtaReadSector(u32 lba, u8 *buffer) {
64 irq14stat = 0;
65 Outb(0x3F6, 0x00); // clear SRST, enable IRQs
66 Outb(ATA_DRIVE, 0xE0 | ((lba >> 24) & 0x0F));
67 for (int i = 0; i < 4; i++) {
68 Inb(0x3F6);
69 }
70 Outb(ATA_SECCNT, 1);
71 Outb(ATA_LBA_LOW, lba & 0xFF);
72 Outb(ATA_LBA_MID, (lba >> 8) & 0xFF);
73 Outb(ATA_LBA_HI, (lba >> 16) & 0xFF);
74 Outb(ATA_COMMAND, ATA_CMD_READ_SECTORS);
75
76 AtaPoll();
77
78 for (int i = 0; i < 256; i++) {
79 u16 data = Inw(ATA_DATA);
80 buffer[i * 2] = (u8)(data & 0xFF);
81 buffer[(i * 2) + 1] = (u8)(data >> 8);
82 }
83}

97void AtaWriteSector(u32 lba, const u8 *buffer) {
98 Outb(ATA_CTRL, 0x00); // enable IRQs
99
100 Outb(ATA_IO_BASE + 6, 0xE0 | ((lba >> 24) & 0x0F)); // drive + LBA mode
101 Outb(ATA_IO_BASE + 2, 1); // sector count
102 Outb(ATA_IO_BASE + 3, (u8)(lba)); // LBA low
103 Outb(ATA_IO_BASE + 4, (u8)(lba >> 8)); // LBA mid
104 Outb(ATA_IO_BASE + 5, (u8)(lba >> 16)); // LBA high
105
106 Outb(ATA_IO_BASE + 7, ATA_WRITE_CMD); // send write command
107
108 // wait for drive to be ready
109 while (!(Inb(ATA_IO_BASE + 7) & 0x08)) {
110 ; // DRQ
111 }
112
113 // write data
114 for (int i = 0; i < 256; i++) {
115 u16 word = ((u16)buffer[(i * 2) + 1] << 8) | buffer[i * 2];
116 Outw(ATA_IO_BASE, word);
117 }
118
119 // Flush cache
120 Outb(ATA_IO_BASE + 7, 0xE7); // flush command
121 while (Inb(ATA_IO_BASE + 7) & 0x80) {
122 ; // wait for BSY to clear
123 }
124}
